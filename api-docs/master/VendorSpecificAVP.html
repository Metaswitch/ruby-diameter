def to_wire
  length_8, length_16 = UInt24.to_u8_and_u16(@content.length + 12)
  avp_flags = @mandatory ? '11000000' : '10000000'
  header = [@code, avp_flags, length_8, length_16, @vendor_id].pack('NB8CnN')
  header + self.padded_content
end
